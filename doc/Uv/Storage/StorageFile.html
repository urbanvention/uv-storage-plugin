def initialize(*args)
  self.options    = {
    :access_level => 'public-read',
    :file_mapping => nil,
    :object => nil
  }
  
  self.raw_file   = args.first if args.first.is_a?(File) or args.first.is_a?(Tempfile)
  self.options.update(args.extract_options!)
  self.options.stringify_keys!
  self.config     = Uv::Storage::Config.new
  self.client     = HTTPClient.new
  self.connection = Uv::Storage::Connection.new(self.config)
  self.object     = self.options['object']
  
  puts "New file initialized #{self.raw_file.class}"
end

def access_level
  retrieve_meta!

end

def exists?(id)
  # find the mapping
  mapping = Uv::Storage::FileMapping.find_by_id(id)
  
  raise MissingFileMapping.new unless mapping.present?
  
  @file = Uv::Storage::StorageFile.new(:file_mapping => mapping)
  
  if @file.read.blank?
    return false
  else
    return true
  end
end

def save
  if self.raw_file.present?
    @result = send_file_to_master!
    
    if self.object
      self.mapping = Uv::Storage::FileMapping.new( 
        :nodes => @result['node_domains'], 
        :file_path => @result['path'], 
        :access_level => @result['access_level'],
        :object_name => self.object.class.to_s.downcase,
        :object_identifier => self.object.id
      )
      
      puts "Saving mapping: #{self.mapping.inspect}"
      
      raise FileObjectMissing.new("#{self.mapping.errors.inspect}") unless self.mapping.valid?
      
      self.mapping.save!
    else
      raise FileObjectMissing.new
    end
  else
    puts "Raw File is missing"
  end
end

def url(expires = nil)
  @urls = []
  
  self.nodes.each do |node|
    @urls << build_url(node)
  end
  
  @urls.shuffle!
  @url = @urls.first
  
  return @url
end
